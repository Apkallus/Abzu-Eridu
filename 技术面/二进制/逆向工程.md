- 查找字符串空字节，计算字符串长度 `strlen()`
    ```c
        // 循环从 flag 指针取4字节，移动指针并寻找空字节
        // 0x00 - 0x01 = 0xff
        // ~ 0x00 = 0xff
        // 0xff & 0xff & 0x80 = 0x80
        7 @ 00400104  int32_t flag_4bytes = *flag_ptr
        8 @ 00400106  ((char*) flag_ptr) += 4
        9 @ 00400114  find_null = (flag_4bytes - 0x1010101) & ~flag_4bytes & 0x80808080
    10 @ 00400119  do while (find_null == 0)
    // 此时 find_null 中的空字节为 0x80
    // 指针领先4字节，在最后一步消掉


    // 定位剩余4字节中空字节在高还是低2字节，将指针移动到空字节后
    // 仍使用 0x80 技巧
    11 @ 00400120  int32_t find_null_4bytes = find_null & 0x8080   
    12 @ 00400125  if (find_null_4bytes == 0)
        13 @ 00400125  find_null u>>= 0x10
        15 @ 0040012c  ((char*) flag_ptr) += 2

    // 在剩余2字节中定位空字节最终位置
    // uk1 未使用，仅做低位字节*2的副作用运算，测试设置 CF 进位标志
    // 0x80 * 2 将设置 CF 进位标志 
    uk1 = (find_null & 0xff)*2)&0xff
    // 0x400c43 为 flag+3
    // 对于领先的4字节指针，若空字节在最后一位则-4，否则-3
    // 得到 flag 字符串长度
    flag_len = ((char*) flag_ptr) - (0x400c43 + CF)
    ```
